    Rewrite moon.yml to add repo:* tasks while preserving existing cargo tasks as backward-compatible aliases.

    Structure the new moon.yml as:

    ```yaml
    # ── Cargo tasks (existing, preserved) ──────────────────────
    tasks:
      build:
        command: 'cargo build --workspace'
        inputs: ['@globs(sources)', 'Cargo.toml', 'Cargo.lock']
      check:
        command: 'cargo check --workspace'
        inputs: ['@globs(sources)', 'Cargo.toml', 'Cargo.lock']
      lint:
        command: 'cargo clippy --workspace -- -D warnings'
        inputs: ['@globs(sources)', 'Cargo.toml', 'Cargo.lock']
      test:
        command: 'cargo test --workspace'
        inputs: ['@globs(sources)', 'Cargo.toml', 'Cargo.lock']
      format:
        command: 'cargo fmt --all -- --check'
        inputs: ['@globs(sources)']
      format-fix:
        command: 'cargo fmt --all'
        inputs: ['@globs(sources)']

    # ── repo:setup ─────────────────────────────────────────────
      repo:setup:
        command: 'bun install && proto install'
        inputs: ['package.json', '.prototools']
      repo:bootstrap:
        deps: ['repo:setup', 'repo:doctor']
      repo:doctor:
        command: |
          echo "=== Toolchain Check ==="
          bun --version || echo "MISSING: bun"
          node --version || echo "MISSING: node"
          cargo --version || echo "MISSING: cargo"
          rustc --version || echo "MISSING: rustc"
          moon --version || echo "MISSING: moon"
          echo "=== Proto Status ==="
          proto status 2>/dev/null || echo "proto not installed"
          echo "=== Env Check ==="
          test -f .env && echo ".env exists" || echo "MISSING: .env"
          echo "DONE"
        inputs: ['.prototools', 'rust-toolchain.toml', '.env.example']
      repo:toolchain-check:
        command: |
          rustc --version && bun --version && moon --version && echo "ALL OK"
        inputs: ['rust-toolchain.toml', '.prototools']

    # ── repo:dev ───────────────────────────────────────────────
      repo:dev-web:
        command: 'cd apps/client/web/app && bun run dev'
        inputs: ['apps/client/web/app/**/*']
      repo:dev-desktop:
        command: 'cd apps/client/native/src-tauri && cargo tauri dev'
        deps: ['repo:dev-api']
        inputs: ['apps/client/native/**/*']
      repo:dev-api:
        command: 'cargo run -p runtime_server'
        deps: ['check']
        inputs: ['servers/api/**/*']
      repo:dev-extension:
        command: 'echo "Extension dev not yet implemented"'
      repo:dev-workers:
        command: 'echo "Workers dev not yet implemented"'
      repo:dev-fullstack:
        deps: ['repo:dev-api', 'repo:dev-web']

    # ── repo:quality ───────────────────────────────────────────
      repo:fmt:
        deps: ['format', 'apps~client~web~app:format']
      repo:lint:
        deps: ['lint', 'apps~client~web~app:lint']
      repo:typecheck:
        deps: ['check', 'apps~client~web~app:check']
      repo:contracts-check:
        command: 'echo "Contracts check — Phase 2 implementation"'
      repo:test-unit:
        deps: ['test', 'apps~client~web~app:test:unit']
      repo:test-integration:
        command: 'cargo test --workspace -- --ignored'
      repo:test-e2e:
        command: 'cd apps/client/web/app && bun run test:e2e'
      repo:test-agent:
        command: 'echo "Agent tests — Phase 5 implementation"'
      repo:verify:
        deps: ['repo:fmt', 'repo:lint', 'repo:typecheck', 'repo:test-unit']

    # ── repo:codegen ───────────────────────────────────────────
      repo:typegen:
        command: 'echo "Typegen — Phase 2 implementation"'
      repo:openapi-gen:
        command: 'echo "OpenAPI gen — Phase 2 implementation"'
      repo:fixtures-gen:
        command: 'echo "Fixtures gen — Phase 5 implementation"'
      repo:icons-gen:
        command: 'echo "Icons gen — as needed"'
      repo:tokens-gen:
        command: 'echo "Tokens gen — as needed"'

    # ── repo:ops ───────────────────────────────────────────────
      repo:trace-open:
        command: 'echo "Trace open — Phase 7 implementation"'
      repo:evals-run:
        command: 'echo "Evals run — Phase 5 implementation"'
      repo:replay-protocol:
        command: 'echo "Replay protocol — Phase 7 implementation"'
      repo:release-dry-run:
        command: 'echo "Release dry run — Phase 9 implementation"'
      repo:release-desktop:
        command: 'echo "Release desktop — Phase 9 implementation"'
      repo:release-web:
        command: 'echo "Release web — Phase 9 implementation"'
      repo:release-server:
        command: 'echo "Release server — Phase 9 implementation"'

    # ── repo:security ──────────────────────────────────────────
      repo:audit-rust:
        command: 'cargo audit 2>/dev/null || echo "Install cargo-audit: cargo install cargo-audit"'
      repo:audit-bun:
        command: 'bun audit || true'
      repo:secrets-scan:
        command: 'echo "Secrets scan — use gitleaks or similar"'
      repo:licenses-check:
        command: 'echo "Licenses check — Phase 9 implementation"'

    # ── Legacy aliases ─────────────────────────────────────────
      lint-all:
        deps: ['repo:lint']
      check-all:
        deps: ['repo:typecheck', 'repo:lint']
      test-all:
        deps: ['repo:test-unit', 'repo:typecheck']
    ```

    Key rules:
    - Preserve existing cargo task names (build, check, lint, test, format, format-fix) — other moon.yml files may reference them
    - repo:* tasks compose existing tasks via deps where possible
    - Placeholder tasks use `echo` with descriptive messages — they'll be implemented in later phases
    - `repo:verify` is the quality gate — fmt + lint + typecheck + test
    - Task inputs should be specific to minimize cache invalidation
