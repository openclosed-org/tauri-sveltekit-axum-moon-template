```typescript
import { test, expect } from '@playwright/test';
import { triggerMockOAuth } from '../fixtures/auth';

// Per D-05: Tenant isolation must have E2E behavior verification (dual coverage)
test.describe('Tenant Isolation (E2E)', () => {
  let tenant1Context: any;
  let tenant2Context: any;

  test.beforeEach(async ({ browser }) => {
    // Set up two browser contexts for two tenants
    const context1 = await browser.newContext();
    const context2 = await browser.newContext();
    tenant1Context = await context1.newPage();
    tenant2Context = await context2.newPage();
  });

  test('user in tenant1 cannot see tenant2 data', async () => {
    // Login as user from tenant1
    await tenant1Context.goto('/login');
    await triggerMockOAuth(tenant1Context, 'tenant1_code');
    await tenant1Context.waitForURL(/\/counter|\/admin/);

    // Navigate to admin/dashboard and check tenant-specific data
    await tenant1Context.goto('/admin');
    // Should only show tenant1's data
    
    // Try accessing tenant2's data - should be empty/blocked
    const response = await tenant1Context.request.get('/api/data', {
      headers: { 'Authorization': 'Bearer tenant1_token' }
    });
    
    // For tenant isolation to work: response should indicate no cross-tenant access
    // Either empty data or explicit denial
    const data = await response.json();
    // Expect: either empty array or filtered results
    expect(data).toBeDefined();
  });

  test('new user signup creates new tenant', async ({ page }) => {
    // Use fresh user for tenant creation test
    await page.goto('/login');
    
    // Trigger OAuth with new user credentials
    await triggerMockOAuth(page, 'new_user_code');
    
    // Verify tenant was created via API
    const response = await page.request.get('/api/tenant/me');
    const tenantData = await response.json();
    
    // Should have tenant_id assigned
    expect(tenantData).toHaveProperty('tenant_id');
    expect(tenantData.tenant_id).toBeTruthy();
  });
});
```

```typescript
import { test, expect } from '@playwright/test';
import { triggerMockOAuth } from '../fixtures/auth';

// Per D-05: Token refresh must have E2E behavior verification
test.describe('Token Refresh (E2E)', () => {
  test('session persists across page refresh', async ({ page }) => {
    // Login first
    await page.goto('/login');
    await triggerMockOAuth(page, 'refresh_test_code');
    await page.waitForURL(/\/counter|\/admin/, { timeout: 10000 });
    
    // Store initial session state
    const initialSession = await page.evaluate(() => localStorage.getItem('session'));
    expect(initialSession).toBeTruthy();
    
    // Refresh page - should maintain session
    await page.reload();
    await page.waitForLoadState('networkidle');
    
    // Should still be logged in, not redirected to login
    const currentUrl = page.url();
    expect(currentUrl).not.toContain('/login');
    
    const sessionAfterRefresh = await page.evaluate(() => localStorage.getItem('session'));
    expect(sessionAfterRefresh).toEqual(initialSession);
  });

  test('token refresh endpoint works before expiry', async ({ page }) => {
    // Login
    await page.goto('/login');
    await triggerMockOAuth(page, 'refresh_test_code');
    await page.waitForURL(/\/counter/);
    
    // Manually trigger token refresh
    const refreshResponse = await page.request.post('/api/auth/refresh', {});
    
    // Should return new token
    expect(refreshResponse.ok()).toBeTruthy();
    
    const newToken = await refreshResponse.json();
    expect(newToken).toHaveProperty('access_token');
  });

  test('auth:expired event triggers re-authentication flow', async ({ page }) => {
    // Login first
    await page.goto('/login');
    await triggerMockOAuth(page, 'expired_test_code');
    await page.waitForURL(/\/counter/);
    
    // Simulate token expiry - dispatch auth:expired event
    // This should trigger the auth store to show re-auth prompt
    await page.evaluate(() => {
      window.dispatchEvent(new CustomEvent('auth:expired'));
    });
    
    // Should show re-authentication prompt or redirect
    // Check for auth expired UI or redirect to login
    const loginButton = page.locator('button:has-text("Sign in"), button:has-text("Sign in with Google")');
    await expect(loginButton.or(page.locator('text=Session expired'))).toBeVisible({ timeout: 5000 });
  });
});
```
