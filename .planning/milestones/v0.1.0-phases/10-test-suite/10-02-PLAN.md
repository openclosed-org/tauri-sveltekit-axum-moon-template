import { describe, it, expect } from 'vitest';

describe('CounterPage', () => {
  it('displays initial counter value of 0', async () => {
    const component = mount(CounterPage);
    const display = component.getByText(/^0$/);
    expect(display).toBeTruthy();
  });

  it('increments counter on increment button click', async () => {
    const component = mount(CounterPage);
    const incrementBtn = component.getByRole('button', { name: /increment/i });
    await incrementBtn.click();
    const display = component.getByText(/^1$/);
    expect(display).toBeTruthy();
  });

  it('decrements counter on decrement button click', async () => {
    const component = mount(CounterPage);
    const decrementBtn = component.getByRole('button', { name: /decrement/i });
    await decrementBtn.click();
    const display = component.getByText(/^-1$/);
    expect(display).toBeTruthy();
  });

  it('resets counter to 0 on reset button click', async () => {
    const component = mount(CounterPage);
    // First increment
    await component.getByRole('button', { name: /increment/i }).click();
    // Then reset
    await component.getByRole('button', { name: /reset/i }).click();
    const display = component.getByText(/^0$/);
    expect(display).toBeTruthy();
  });
});
