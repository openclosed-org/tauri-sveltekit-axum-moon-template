```typescript
    await use();
  }

// Helper to trigger mock OAuth deep-link
export async function triggerMockOAuth(page: Page, mockCode: string = 'mock_auth_code') {
```

```typescript
test.describe('Login Flow (E2E)', () => {
  test('shows login page with Google sign-in button', async ({ page }) => {
    await page.goto('/login');
    
    // Check for Google sign-in button presence
    const signInBtn = page.getByRole('button', { name: /sign in with google/i });
    await expect(signInBtn).toBeVisible();
    
    // Check page has proper layout
    await expect(page.locator('h1, [role="heading"]')).toContainText(/login/i);
  });

  test('mock OAuth callback logs user in via deep-link', async ({ page }) => {
    await page.goto('/login');
    
    // Trigger mock deep-link OAuth callback
    // This tests the callback handling chain in +layout.svelte
    await triggerMockOAuth(page, 'test_mock_code');
    
    // After callback processed, should be logged in
    // Check for sign out button or user info display
    await expect(page.locator('button:has-text("Sign out")')).toBeVisible({ timeout: 10000 });
  });

  test('already authenticated user redirects to counter', async ({ page }) => {
    // Pre-condition: User already has session (simulate)
    await page.goto('/login');
    
    // If session exists, should redirect to counter page
    // Check URL or content indicates counter page
    await page.waitForLoadState('networkidle');
    const url = page.url();
    // Either stays on login (not authenticated) or redirects to counter
    expect(url.includes('counter') || url.includes('login')).toBeTruthy();
  });

  test('login page is responsive on mobile', async ({ page, viewport }) => {
    await page.setViewportSize({ width: 375, height: 667 });
    await page.goto('/login');
    
    // Login page should still be usable on mobile

    await expect(signInBtn).toBeVisible();
  });
});
```

```typescript
import { test, expect } from '@playwright/test';

test.describe('Counter Page (E2E)', () => {
  test.beforeEach(async ({ page }) => {
    // Navigate to counter page (assumes user is logged in)
    // In real scenario would use authenticated session
    await page.goto('/counter');
  });

  test('displays counter with initial value 0', async ({ page }) => {
    const counterDisplay = page.getByText(/^0$/);
    await expect(counterDisplay).toBeVisible();
  });

  test('increment button increases count', async ({ page }) => {
    await page.getByRole('button', { name: /increment/i }).click();
    const counterDisplay = page.getByText(/^1$/);
    await expect(counterDisplay).toBeVisible();
  });

  test('decrement button decreases count', async ({ page }) => {
    await page.getByRole('button', { name: /decrement/i }).click();
    const counterDisplay = page.getByText(/^-1$/);
    await expect(counterDisplay).toBeVisible();
  });

  test('reset button returns to 0', async ({ page }) => {
    // First increment
    await page.getByRole('button', { name: /increment/i }).click();
    await page.getByRole('button', { name: /increment/i }).click();
    
    // Then reset
    await page.getByRole('button', { name: /reset/i }).click();
    const counterDisplay = page.getByText(/^0$/);
    await expect(counterDisplay).toBeVisible();
  });
});
```
